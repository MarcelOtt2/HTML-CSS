  <script>
    const sounds = [
      { key: 'Q', name: 'Heater 1', file: 'Heater-1.mp3' },
      { key: 'W', name: 'Heater 2', file: 'Heater-2.mp3' },
      { key: 'E', name: 'Heater 3', file: 'Heater-3.mp3' },
      { key: 'A', name: 'Heater 4', file: 'Heater-4_1.mp3' },
      { key: 'S', name: 'Clap', file: 'Heater-6.mp3' },
      { key: 'D', name: 'Open-HH', file: 'Dsc_Oh.mp3' },
      { key: 'Z', name: 'Kick-n\'-Hat', file: 'Kick_n_Hat.mp3' },
      { key: 'X', name: 'Kick', file: 'RP4_KICK_1.mp3' },
      { key: 'C', name: 'Closed-HH', file: 'Cev_H2.mp3' }
    ];

    const display = document.getElementById('display');
    const padBank = document.getElementById('pad-bank');

    // Create drum pads dynamically
    sounds.forEach(sound => {
      const pad = document.createElement('div');
      pad.className = 'drum-pad';
      pad.id = sound.name.replace(/\s+/g, '-'); // e.g., "Heater-1"
      pad.textContent = sound.key;

      const audio = document.createElement('audio');
      audio.className = 'clip';
      audio.id = sound.key;
      audio.src = `https://cdn.freecodecamp.org/curriculum/drum/${sound.file}`;

      pad.appendChild(audio);
      padBank.appendChild(pad);

      // Click handler
      pad.addEventListener('click', () => playSound(sound.key, sound.name));

      // Optional visual feedback
      audio.addEventListener('play', () => pad.classList.add('playing'));
      audio.addEventListener('ended', () => pad.classList.remove('playing'));
    });

    // Keyboard handler
    document.addEventListener('keydown', (e) => {
      const key = e.key.toUpperCase();
      const sound = sounds.find(s => s.key === key);
      if (sound) {
        playSound(key, sound.name);
      }
    });

    function playSound(key, name) {
      const audio = document.getElementById(key);
      if (audio) {
        // Reset current time to allow rapid re-trigger
        audio.currentTime = 0;
        audio.play().catch(e => console.warn('Audio play interrupted:', e));
        display.textContent = name;
      }
    }
  </script>
